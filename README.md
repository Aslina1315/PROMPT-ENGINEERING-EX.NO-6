Python Code for Multi-AI Integration
import requests

# Prompt for AI tools
prompt = "As a programmer, suggest Python techniques to improve code efficiency."

# Example API endpoints (placeholders)
chatgpt_api = "https://api.openai.com/v1/chat/completions"
gemini_api = "https://api.gemini.google.com/v1/generate"

headers = {
    "Content-Type": "application/json"
}

data = {
    "prompt": prompt,
    "max_tokens": 100
}

# Function to call AI API
def get_ai_response(api_url):
    response = requests.post(api_url, headers=headers, json=data)
    if response.status_code == 200:
        return response.json()
    else:
        return "Error in API request"

# Collect responses
chatgpt_output = get_ai_response(chatgpt_api)
gemini_output = get_ai_response(gemini_api)

# Display outputs
print("ChatGPT Response:")
print(chatgpt_output)

print("\nGemini Response:")
print(gemini_output)
